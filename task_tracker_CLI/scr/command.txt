# Adding a new task
task_tacker add "Buy groceries"
# Output: Task added successfully (ID: 1)

# Updating and deleting tasks
task_tacker update 1 "Buy groceries and cook dinner"
task_tacker delete 1

# Marking a task as in progress or done
task_tacker mark-in-progress 1
task_tacker mark-done 1

# Listing all tasks
task_tacker list

# Listing tasks by status
task_tacker list done
task_tacker list todo
task_tacker list in-progress
